country_event = {
    id = dsi.1
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        is_ai = no
        NOT = { has_country_flag = dsi_incident_scheduled }
        NOT = { has_country_flag = dsi_recent_incident }
    }

    immediate = {
        if = {
            limit = { NOT = { has_country_flag = dsi_department_active } }
            set_timed_country_flag = { flag = dsi_incident_scheduled days = 365 }
            country_event = { id = dsi.100 days = 1 random = 364 }
        }
        else = {
            set_timed_country_flag = { flag = dsi_incident_scheduled days = 365 }
            country_event = { id = dsi.101 days = 1 random = 364 }
        }
    }
}
