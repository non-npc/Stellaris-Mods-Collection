decision_pf_unity_festival = {
    owned_planets_only = yes
    icon = decision_pf_unity_festival

    resources = {
        category = decisions
        cost = {
            unity = 300
            energy = 200
        }
    }

    potential = {
        owner = { has_technology = tech_pf_cultural_festival_coordination }
    }

    allow = {
        custom_tooltip = {
            fail_text = pf_requires_no_active_festival
            NOT = { pf_has_active_planetary_festival = yes }
        }
    }

    effect = {
        add_modifier = {
            modifier = pf_unity_festival_modifier
            days = 1800
        }
    }

    ai_weight = { weight = 5 }
}
