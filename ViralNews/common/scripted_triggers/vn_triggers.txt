vn_has_hypernet_nexus = {
	OR = {
		any_owned_planet = {
			is_colony = yes
			has_building = building_vn_hypernet_media_nexus
		}
		any_owned_starbase = {
			is_arkship_starbase = yes
			has_starbase_building = vn_arkship_hypernet_media_nexus
		}
		any_owned_fleet = {
			starbase? = {
				is_arkship_starbase = yes
				has_starbase_building = vn_arkship_hypernet_media_nexus
			}
		}
	}
}

vn_can_receive_news = {
	is_ai = no
	has_country_flag = vn_network_active
	NOT = { has_country_flag = vn_network_activation_pending }
	NOT = { has_country_flag = vn_news_cooldown }
	vn_has_hypernet_nexus = yes
}
