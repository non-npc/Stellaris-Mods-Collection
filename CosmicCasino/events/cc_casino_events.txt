	id = cc.20
	title = cc.wager_name_low_slots
	desc = cc.confirm_low_slots.desc
	picture = GFX_evt_cc_slots
	is_triggered_only = yes

	option = {
		name = cc.place_wager
		custom_tooltip = cc.wager_uses_available
		trigger = { owner = { has_resource = { type = energy amount > 0 } } }

# Hidden wager resolvers. These perform the payout/loss after 30 days, then open the visible result event with the correct sound.
colony_event = {
	id = cc.101
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		remove_modifier = cc_casino_transaction_pending
		owner = {
			remove_country_flag = cc_casino_bonus_payout
			set_variable = { which = cc_casino_bonus_alloys value = 0 }
			set_variable = { which = cc_casino_bonus_unity value = 0 }
		}
		random_list = {
			45 = {
				owner = { cc_record_loss = yes }
				colony_event = { id = cc.201 }
			}
			35 = {
				owner = { cc_award_energy_win = { BASE = 750 } }
				colony_event = { id = cc.203 }
			}
			15 = {
				owner = { cc_award_energy_win = { BASE = 1500 } }
				colony_event = { id = cc.204 }
			}
			4 = {
				owner = {
					cc_award_energy_win = { BASE = 3000 }
					add_modifier = { modifier = cc_lucky_streak days = 1800 }
				}
				colony_event = { id = cc.204 }
			}
			1 = {
				owner = {
					cc_award_energy_win = { BASE = 10000 }
					add_modifier = { modifier = cc_lucky_streak days = 3600 }
				}
				colony_event = { id = cc.205 }
			}
		}
	}
}
