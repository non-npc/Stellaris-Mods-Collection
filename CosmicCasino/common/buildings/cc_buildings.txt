	base_buildtime = 480
	category = amenity

	building_sets = {
		government
		urban

	}

	icon = building_cc_cosmic_casino

	potential = {
		exists = owner
		owner = {
			OR = {
				has_tradition = tr_cc_casino_finish
				has_country_flag = cc_cosmic_gaming_finished
			}
		}
	}

	allow = {
		exists = owner

			OR = {
				has_tradition = tr_cc_casino_finish
				has_country_flag = cc_cosmic_gaming_finished
			}
			custom_tooltip = {
				fail_text = cc_requires_no_existing_cosmic_casino

		}
	}

		planet_amenities_add = 15
		pop_happiness = 0.03
	}

	triggered_planet_modifier = {
		potential = {
			always = yes
		}
		modifier = {
			job_cc_casino_dealer_add = 2
			job_cc_casino_manager_add = 1
		}
	}

	ai_weight = {
		weight = 0
	}

	on_built = {
		if = {
			limit = {
				exists = owner
				owner = { is_ai = no }
			}

		}
	}
}
