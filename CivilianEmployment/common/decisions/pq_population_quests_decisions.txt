decision_pq_adventure_quests = {
    owned_planets_only = yes
    sound = event_cityscape
    icon = decision_expel_population

    resources = {
        category = decisions
        cost = {
            influence = 50
        }
    }

    prerequisites = {
        tech_pq_population_quests
    }

    potential = {
        exists = owner
        owner = { is_ai = no }
        NOT = { has_modifier = pq_modifier_adventure_guild }
    }

    allow = {
        exists = owner
        owner = { is_ai = no }
        NOT = { has_modifier = pq_modifier_adventure_guild }
    }

    effect = {
        custom_tooltip = decision_pq_adventure_quests_effect
        add_modifier = { modifier = "pq_modifier_adventure_guild" days = -1 }

        hidden_effect = {
            remove_modifier = pq_modifier_virtual_forces
            remove_modifier = pq_modifier_free_thinkers
        }
    }

    ai_weight = {
        weight = 0
    }
}
