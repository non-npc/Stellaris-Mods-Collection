fw_invoke_fate_button_effect = {
	potential = {
		OR = {
			is_scope_type = planet
			is_scope_type = ship
		}
		from = {
			has_technology = tech_fateweaving
		}
		colony = {
			is_colony = yes
		NOT = { has_carrier_flag = fw_fate_pending }
		NOT = { has_carrier_flag = fw_fate_result_active }
		}
	}
	allow = {
		custom_tooltip = {
			success_text = fw_invoke_fate_button_tooltip
			always = yes
		}
	}
	effect = {
		colony = {
			save_event_target_as = fw_target_colony
		}
		from = {
			save_event_target_as = fw_invoker_country
			country_event = { id = fw.10 }
		}
	}
}
