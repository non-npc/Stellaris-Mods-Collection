namespace = fw

country_event = {
	id = fw.10
	title = fw.10.name
	desc = fw.10.desc
	picture = GFX_fw_evt_marked
	show_sound = fw_markcolony
	location = event_target:fw_target_colony
	is_triggered_only = yes

	trigger = {
		exists = event_target:fw_target_colony
		exists = event_target:fw_invoker_country
		event_target:fw_invoker_country = {
			has_technology = tech_fateweaving
		}
		event_target:fw_target_colony = {
			is_colony = yes
			NOT = { has_carrier_flag = fw_fate_pending }
			NOT = { has_carrier_flag = fw_fate_result_active }
		}
	}

	option = {
		name = fw.10.a
		hidden_effect = {
			country_event = { id = fw.11 }
		}
	}

	option = {
		name = fw.10.b
		default_hide_option = yes
	}
}

country_event = {
	id = fw.11
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		exists = event_target:fw_target_colony
		event_target:fw_target_colony = {
			is_colony = yes
			NOT = { has_carrier_flag = fw_fate_pending }
			NOT = { has_carrier_flag = fw_fate_result_active }
		}
	}

	immediate = {
		event_target:fw_target_colony = {
			set_carrier_flag = fw_fate_pending
			colony_event = { id = fw.100 days = 30 random = 60 }
		}
	}
}

colony_event = {
	id = fw.100
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_carrier_flag = fw_fate_pending
	}

	immediate = {
		save_event_target_as = fw_target_colony
		from = {
			save_event_target_as = fw_invoker_country
			random_list = {
				1 = { country_event = { id = fw.101 } }
				1 = { country_event = { id = fw.102 } }
				1 = { country_event = { id = fw.103 } }
